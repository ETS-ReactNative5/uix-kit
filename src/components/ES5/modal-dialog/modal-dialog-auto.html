Modal Dialog Automatically Open

Use the following script to fire it.

( function( $ ) {
"use strict";
	$( document ).ready( function() {
		$(document).UixFireModalDialog({
			id: 'modal-open-1',
			height: false,
			width: false,
			speed: 500,
			lightbox  : true,
			autoClose : false
		});
	} );
} ) ( jQuery );
